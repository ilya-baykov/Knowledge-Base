# Выбираем базоый образ
FROM python:3.11.14-slim

# Устанавливаем рабочую директорию внутри контейнера
WORKDIR /app

# Копируем файл с зависимостями в контейнер
COPY requirements.txt .

# Устанавливаем зависимости
RUN pip install --no-cache-dir -r requirements.txt

# Копируем исходный код приложения в контейнер ( предполагается, что код находится в папке src, можно просто COPY . . )
# Не забывать про файл .dockerignore чтобы не копировать лишние файлы
COPY src/ ./src/

# Указываем команду для запуска приложения
CMD ["python", "src/main.py"]








# ДОПОЛНИТЕЛЬНО

Если нужно видеть логи при интерактивном запуске (docker run -i ... )
Можно добавить :
ENV PYTHONUNBUFFERED=1 (Устанавливаем переменную окружения для Python (чтобы вывод был не буферизован))



+ Лучше создать отдельного пользователя для работы внутри контейнера

RUN groupadd -r groupdocker && useradd -r -g groupdocker userdocker   - создать группу и пользователя
USER userdocker   - выбор созданного пользователя






EXPOSE 8000 - Метка порта ( только информационная нагрузка, как label )
