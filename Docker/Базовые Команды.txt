docker images - Посмотреть все образы

docker  pull <name_images>   - выгрузить образ из DockerHub


docker run <name_images>     - создать и запустить контейнер с нужным образом
docker run --name nginx80 -d -p 8000:80  ca871a86d45a  ( Пример с пробросом порта, где 8000-порт хоста, а 80 - порт контейнера 

docker run -it web_backend:latest /bin/bash   ( Пример запуска для просмотра содержимого папок внутри контейнера ( до запуска всех команд ))
docker exec -it c679c5d74489 psql -U candidates_chat    ( Пример уже запущенного кантейнера )

docker ps  - Посмотреть все ЗАПУЩЕННЫЕ контейнеры или docker ps -a ( Посмотреть все контейенеры,в том числе и те что уже отработались )

docker start <name_images>/<container_id> - запустить уже созданный контейнер ( есть флаг -i для интерактивности)

docker stop <name_images>/<container_id> - остановить контейнер 
docker kill <name_images>/<container_id> - ЖЕСКАЯ ОСТАНОВКА КОНТЕЙНЕРА ( если stop не работает )


docker rm  <name_container>/<container_id> - Удалить контейнер
docker rmi  <name_images> - Удалить образ ( важно правильно указать тег образа )


docker <name_func> --help - Просмотр всех доступных флагов для функции


docker build <Dockerfile_path> -t <name_image>


docker container inspect <container_id>  - Просмотр всей информации по контейнеру

docker port <container_id>   - Посмотреть порт 


docker compose up    - запускает мультиконтейнерное приложение, описанное в файле docker-compose.yml, создавая и запуская все его сервисы (контейнеры), сети, тома и конфигурации одним действием, что «поднимает» весь стек приложения
docker compose up --build  ( лучше с флагом, чтобы сделать 'пересборку' после изменений в коде
docker compose stop  - останавливает запущенный контейнер
docker compose down  - останавливает и удаляет запущенный контейнер  ( образы остаются )
