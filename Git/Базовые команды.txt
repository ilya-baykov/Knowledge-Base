git init — инициализировать новый репозиторий в текущей папке.

git clone <url-репозитория> — склонировать (скачать) удалённый репозиторий по ссылке.

git status — показать состояние рабочего каталога и индекса (какие файлы изменены, но не добавлены, какие добавлены и готовы к коммиту).

git add <файл> — добавить конкретный файл в индекс.

git commit -m "Сообщение коммита" — зафиксировать изменения, сохранённые в индексе, с понятным сообщением.


git diff — показать изменения в файлах, которые еще не добавлены в индекс.

git log — показать историю коммитов в текущей ветке.


git branch — показать список локальных веток (текущая будет выделена *).

git checkout -b <имя-ветки> или git switch -c <имя-ветки> — создать новую ветку и сразу переключиться на неё.

git merge <имя-ветки> — влить указанную ветку в текущую.

git branch -d <имя-ветки> — удалить локальную ветку (если она уже слита).



git remote -v — показать список удалённых репозиториев, привязанных к локальному.

git remote add origin <url> — привязать удалённый репозиторий с именем origin (обычное имя для основного удалённого репо).

git push origin <имя-ветки> — отправить (запушить) локальную ветку на удалённый репозиторий origin.

git pull — скачать изменения из привязанной удалённой ветки и автоматически слить (merge) с текущей локальной веткой. Перед pull часто делают git commit своих изменений.

git fetch — только скачать изменения из удалённого репозитория, но НЕ сливать их. Позволяет посмотреть, что изменилось.



git restore <файл> — отменить изменения в рабочем каталоге для конкретного (еще не добавленного в индекс) файла (Git 2.23+).


git stash — временно «спрятать» незакоммиченные изменения и вернуть рабочую директорию к последнему коммиту.

git stash pop — вернуть последние «спрятанные» изменения обратно.



git rm -r --cached .          - Удалить все файлы из индекса ( очистит кеш гита )
